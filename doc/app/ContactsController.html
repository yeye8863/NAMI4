def create
    a = params[:attr]
    @donor = Donor.find(params[:id]);
    @contact = @donor.contacts.build({
        :contact_date => a[0],
        :followup_date => a[1],
        :narrative => a[2]
    })
    @contact.save!
    render :json => @contact if request.xhr?
end
